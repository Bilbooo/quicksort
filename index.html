QuickSort(A, p, r)
 if (p < r) {
  q = Partition
       (A, p, r)
  QuickSort
    (A, p, q-1)
  QuickSort
    (A, q+1, r)
 }

Partition(A, p, r)
 x = A[r]  ← pivot
 i = p − 1
 for j = p to r-1:
  if A[j] ≤ x:
   i++
   swap(A[i], A[j])
 swap(A[i+1], A[r])
 return i + 1
